<%

  "==========================================================================

  " invoice_form.htm

  " Handler utama untuk Park Invoice ke SAP via BAPI_INCOMINGINVOICE_PARK.

  "==========================================================================



  "========== TYPE DEFINITIONS ==========

  TYPES: BEGIN OF ty_item_parsed,

           invoice_doc_item TYPE n LENGTH 6,

           po_number        TYPE ekko-ebeln,

           po_item          TYPE ekpo-ebelp,

           bukrs            TYPE ekko-bukrs,

           quantity         TYPE ekbe-menge,

           po_unit          TYPE ekpo-meins,

           item_amount      TYPE bapiwrbtr,

           ref_doc          TYPE ekbe-belnr,

           ref_doc_year     TYPE ekbe-gjahr,

           ref_doc_it       TYPE ekbe-buzei,

           tax_code         TYPE mwskz,

         END OF ty_item_parsed.

  TYPES tt_items_parsed TYPE STANDARD TABLE OF ty_item_parsed WITH EMPTY KEY.



  " Struktur tambahan untuk UI Material Items

  TYPES: BEGIN OF ty_ui_item,

           itm        TYPE string,

           po_number  TYPE ebeln,

           material   TYPE matnr,

           value_str  TYPE c LENGTH 30, " <-- SUDAH DIBETULKAN MENJADI NATIVE CHAR

           value_num  TYPE wrbtr,

         END OF ty_ui_item.



  "========== VARIABLES ==========

  DATA: lv_action      TYPE string,

        gv_active_page TYPE string,

        lv_inv_date    TYPE dats,

        lv_pstng_date  TYPE dats,

        lv_bsl_date    TYPE dats,

        lv_ref_doc_no  TYPE xblnr1,

        lv_gross_amt   TYPE wrbtr,

        lv_tax_amt     TYPE wrbtr,

        lv_calc_tax    TYPE xfeld,

        lv_pmt_block   TYPE dzlspr,

        lv_header_txt  TYPE string,

        lv_items_json  TYPE string,

        lv_attachments TYPE string,

        lv_comp_code   TYPE bukrs,

        lv_success     TYPE abap_bool,

        lv_processed   TYPE abap_bool,

        lv_docnum      TYPE belnr_d,

        lv_fyear       TYPE gjahr,

        lv_errmsg      TYPE string,

        lt_items_parsed TYPE tt_items_parsed.



  DATA: lv_vendor      TYPE lifnr,

        lv_name1       TYPE name1_gp,

        lv_banka       TYPE banka,

        lv_bankn       TYPE bankn,

        lv_koinh       TYPE lfbk-koinh, 

        lt_ui_items    TYPE TABLE OF ty_ui_item,

        ls_ui_item     TYPE ty_ui_item,

        lv_items_tot   TYPE wrbtr VALUE 0,

        lv_nett        TYPE wrbtr,

        lv_items_str   TYPE c LENGTH 30,

        lv_tax_str     TYPE c LENGTH 30,

        lv_nett_str    TYPE c LENGTH 30,

        lv_invdt_str   TYPE string,

        lv_pstngdt_str TYPE string,

        lv_duedt_str   TYPE string.



  "========== RETRIEVE INPUT ==========

  lv_action      = request->get_form_field( 'action' ).

  lv_inv_date    = request->get_form_field( 'inv_date' ).

  lv_pstng_date  = request->get_form_field( 'pstng_date' ).

  lv_ref_doc_no  = request->get_form_field( 'ref_doc_no' ).

  lv_gross_amt   = request->get_form_field( 'gross_amount' ).

  lv_tax_amt     = request->get_form_field( 'tax_amount' ).

  lv_calc_tax    = request->get_form_field( 'calc_tax_ind' ).

  lv_pmt_block   = request->get_form_field( 'pmt_block' ).

  lv_header_txt  = request->get_form_field( 'header_txt' ).

  lv_items_json  = request->get_form_field( 'items_json' ).

  

  " Baseline Date fallback

  lv_bsl_date = request->get_form_field( 'bsl_date' ).

  IF lv_bsl_date IS INITIAL.

    lv_bsl_date = lv_inv_date.

  ENDIF.



  "========== FLOW LOGIC - PARK INVOICE ==========

  IF lv_action = 'save'.



    lv_processed = abap_true.



    "---------- Validasi dasar ----------

    IF lv_inv_date IS INITIAL.

      lv_success = abap_false.

      lv_errmsg  = 'Tanggal Invoice wajib diisi.'.

    ELSEIF lv_ref_doc_no IS INITIAL.

      lv_success = abap_false.

      lv_errmsg  = 'Nomor Referensi / Nomor Faktur wajib diisi (digunakan untuk deteksi double invoice).'.

    ELSEIF lv_gross_amt IS INITIAL.

      lv_success = abap_false.

      lv_errmsg  = 'Gross Amount wajib diisi.'.

    ELSEIF lv_items_json IS INITIAL.

      lv_success = abap_false.

      lv_errmsg  = 'Tidak ada item GR yang dipilih. Silakan kembali dan centang minimal satu baris.'.

    ELSE.



      "---------- Parse JSON items ----------

      DATA: lv_json_work  TYPE string,

            lv_obj_str    TYPE string,

            lv_key        TYPE string,

            lv_val        TYPE string,

            lv_pos_start  TYPE i,

            lv_pos_end    TYPE i,

            lv_pos_colon  TYPE i,

            ls_item_p     TYPE ty_item_parsed,

            lt_pairs      TYPE TABLE OF string,

            lv_pair       TYPE string.



      lv_json_work = lv_items_json.

      REPLACE FIRST OCCURRENCE OF '[' IN lv_json_work WITH ''.

      REPLACE ALL  OCCURRENCES  OF ']' IN lv_json_work WITH ''.

      CONDENSE lv_json_work.



      WHILE lv_json_work IS NOT INITIAL.

        lv_pos_start = find( val = lv_json_work sub = '{' ).

        lv_pos_end   = find( val = lv_json_work sub = '}' ).



        IF lv_pos_start < 0 OR lv_pos_end < 0.

          EXIT.

        ENDIF.



        lv_obj_str = substring( val = lv_json_work off = lv_pos_start + 1

                                len = lv_pos_end - lv_pos_start - 1 ).



        lv_json_work = substring( val = lv_json_work

                                  off = lv_pos_end + 1

                                  len = strlen( lv_json_work ) - lv_pos_end - 1 ).



        CLEAR ls_item_p.

        CLEAR lt_pairs.

        SPLIT lv_obj_str AT ',' INTO TABLE lt_pairs.

        

        LOOP AT lt_pairs INTO lv_pair.

          lv_pos_colon = find( val = lv_pair sub = ':' ).

          IF lv_pos_colon < 0. CONTINUE. ENDIF.



          lv_key = substring( val = lv_pair off = 0 len = lv_pos_colon ).

          lv_val = substring( val = lv_pair off = lv_pos_colon + 1

                              len = strlen( lv_pair ) - lv_pos_colon - 1 ).



          REPLACE ALL OCCURRENCES OF '"' IN lv_key WITH ''.

          REPLACE ALL OCCURRENCES OF '"' IN lv_val WITH ''.

          CONDENSE lv_key.

          CONDENSE lv_val.



          CASE lv_key.

            WHEN 'invoice_doc_item'. ls_item_p-invoice_doc_item = lv_val.

            WHEN 'po_number'.        ls_item_p-po_number        = lv_val.

            WHEN 'po_item'.          ls_item_p-po_item          = lv_val.

            WHEN 'bukrs'.            ls_item_p-bukrs            = lv_val.

            WHEN 'quantity'.         ls_item_p-quantity         = lv_val.

            WHEN 'po_unit'.          ls_item_p-po_unit          = lv_val.

            WHEN 'item_amount'.      ls_item_p-item_amount      = lv_val.

            WHEN 'ref_doc'.          ls_item_p-ref_doc          = lv_val.

            WHEN 'ref_doc_year'.     ls_item_p-ref_doc_year     = lv_val.

            WHEN 'ref_doc_it'.       ls_item_p-ref_doc_it       = lv_val.

            WHEN 'tax_code'.         ls_item_p-tax_code         = lv_val.

          ENDCASE.

        ENDLOOP.



        IF ls_item_p-po_number IS NOT INITIAL.

          APPEND ls_item_p TO lt_items_parsed.

          IF lv_comp_code IS INITIAL.

            lv_comp_code = ls_item_p-bukrs.

          ENDIF.

        ENDIF.



      ENDWHILE.



      IF lt_items_parsed IS INITIAL.

        lv_success = abap_false.

        lv_errmsg  = 'Gagal mem-parse data item. Tidak ada item valid yang ditemukan.'.

      ELSE.



        "========== AMBIL LAMPIRAN DARI ZAPP_ATTACHMENT ==========

        TYPES: BEGIN OF ty_po,

                 ebeln TYPE ebeln,

               END OF ty_po.

        DATA: lt_po_list TYPE TABLE OF ty_po,

              ls_po_list TYPE ty_po.



        DATA: lr_ref_doc TYPE RANGE OF ebeln,

              ls_ref_doc LIKE LINE OF lr_ref_doc.



        LOOP AT lt_items_parsed ASSIGNING FIELD-SYMBOL(<ls_pitem>).

          IF <ls_pitem>-po_number IS NOT INITIAL.

            ls_po_list-ebeln = <ls_pitem>-po_number.

            CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'

              EXPORTING input  = ls_po_list-ebeln

              IMPORTING output = ls_po_list-ebeln.

            COLLECT ls_po_list INTO lt_po_list.

          ENDIF.

        ENDLOOP.



        LOOP AT lt_po_list INTO ls_po_list.

          CLEAR ls_ref_doc.

          ls_ref_doc-sign   = 'I'.

          ls_ref_doc-option = 'EQ'.

          ls_ref_doc-low    = ls_po_list-ebeln.

          APPEND ls_ref_doc TO lr_ref_doc.

        ENDLOOP.



        CLEAR lv_attachments.

        lv_attachments = '['.



        IF lr_ref_doc IS NOT INITIAL.

          DATA: lt_db_attach TYPE TABLE OF zapp_attachment.

          SELECT * FROM zapp_attachment

            WHERE ref_doc IN @lr_ref_doc

            INTO TABLE @lt_db_attach.



          DATA(lv_first_att) = abap_true.

          LOOP AT lt_db_attach INTO DATA(ls_db_attach).

            IF lv_first_att = abap_true.

              lv_first_att = abap_false.

            ELSE.

              lv_attachments = lv_attachments && ','.

            ENDIF.



            DATA(lv_b64) = cl_http_utility=>encode_x_base64( unencoded = ls_db_attach-file_content ).

            DATA(lv_fname) = ls_db_attach-file_name.

            REPLACE ALL OCCURRENCES OF '"' IN lv_fname WITH '\"'.

            

            DATA(lv_mtype) = ls_db_attach-file_type.

            IF lv_mtype IS INITIAL. lv_mtype = 'image/jpeg'. ENDIF.



            lv_attachments = lv_attachments &&

              `{"type":"` && ls_db_attach-doc_type && `",` &&

              `"name":"` && lv_fname && `",` &&

              `"mime":"` && lv_mtype && `",` &&

              `"content":"` && lv_b64 && `"}`.

          ENDLOOP.

        ENDIF.

        lv_attachments = lv_attachments && ']'.



        "---------- Bangun HEADERDATA ----------

        DATA: ls_headerdata   TYPE bapi_incinv_create_header,

              ls_itemdata     TYPE bapi_incinv_create_item,

              lt_itemdata     TYPE TABLE OF bapi_incinv_create_item,

              ls_taxdata      TYPE bapi_incinv_create_tax,

              lt_taxdata      TYPE TABLE OF bapi_incinv_create_tax,

              lt_return       TYPE bapiret2_t,

              lv_invdocnumber TYPE belnr_d,

              lv_fiscalyear   TYPE gjahr,

              lv_has_error    TYPE abap_bool,

              lv_msg          TYPE string.



        ls_headerdata-invoice_ind  = 'X'.

        ls_headerdata-doc_type     = 'RE'.

        ls_headerdata-doc_date     = lv_inv_date.

        ls_headerdata-pstng_date   = lv_pstng_date.

        ls_headerdata-comp_code    = lv_comp_code.

        ls_headerdata-gross_amount = lv_gross_amt.



        DATA lv_waers TYPE ekko-waers.

        lv_waers = request->get_form_field( 'waers' ).

        IF lv_waers IS INITIAL. lv_waers = 'IDR'. ENDIF.

        ls_headerdata-currency     = lv_waers.

        ls_headerdata-calc_tax_ind = lv_calc_tax.

        ls_headerdata-ref_doc_no   = lv_ref_doc_no.

        ls_headerdata-header_txt   = lv_header_txt.

        ls_headerdata-bline_date   = lv_bsl_date.



        "---------- Bangun TAXDATA ----------

        IF lv_calc_tax IS INITIAL AND lv_tax_amt IS NOT INITIAL AND lv_tax_amt <> '0'.

          READ TABLE lt_items_parsed INDEX 1 INTO DATA(ls_first_item).

          IF sy-subrc = 0.

            ls_taxdata-tax_code = ls_first_item-tax_code.

          ENDIF.

          ls_taxdata-tax_amount      = lv_tax_amt.

          ls_taxdata-tax_base_amount = lv_gross_amt - lv_tax_amt.

          APPEND ls_taxdata TO lt_taxdata.

        ENDIF.



        "---------- Bangun ITEMDATA ----------

        LOOP AT lt_items_parsed ASSIGNING FIELD-SYMBOL(<ls_ip>).

          CLEAR ls_itemdata.

          ls_itemdata-invoice_doc_item = <ls_ip>-invoice_doc_item.

          ls_itemdata-po_number        = <ls_ip>-po_number.

          ls_itemdata-po_item          = <ls_ip>-po_item.

          ls_itemdata-quantity         = <ls_ip>-quantity.

          ls_itemdata-po_unit          = <ls_ip>-po_unit.

          ls_itemdata-item_amount      = <ls_ip>-item_amount.

          ls_itemdata-ref_doc          = <ls_ip>-ref_doc.

          ls_itemdata-ref_doc_year     = <ls_ip>-ref_doc_year.

          ls_itemdata-ref_doc_it       = <ls_ip>-ref_doc_it.



          IF <ls_ip>-tax_code IS NOT INITIAL.

            ls_itemdata-tax_code       = <ls_ip>-tax_code.

          ENDIF.

          APPEND ls_itemdata TO lt_itemdata.

        ENDLOOP.



        "---------- Panggil BAPI ----------

        CALL FUNCTION 'BAPI_INCOMINGINVOICE_PARK'

          EXPORTING

            headerdata       = ls_headerdata

          IMPORTING

            invoicedocnumber = lv_invdocnumber

            fiscalyear       = lv_fiscalyear

          TABLES

            itemdata         = lt_itemdata

            taxdata          = lt_taxdata

            return           = lt_return.



        "---------- Cek Return Messages ----------

        LOOP AT lt_return ASSIGNING FIELD-SYMBOL(<ls_ret>) WHERE type = 'E' OR type = 'A'.

          lv_has_error = abap_true.

          IF lv_msg IS INITIAL. lv_msg = <ls_ret>-message. ELSE. lv_msg = lv_msg && ' | ' && <ls_ret>-message. ENDIF.

        ENDLOOP.



        IF lv_has_error = abap_false.

          CALL FUNCTION 'BAPI_TRANSACTION_COMMIT' EXPORTING wait = 'X'.

          lv_success = abap_true.

          lv_docnum  = lv_invdocnumber.

          lv_fyear   = lv_fiscalyear.



          "====================================================================

          " FETCH DATA UNTUK UI DOCUMENT DETAIL SETELAH PARK SUCCESS

          "====================================================================

          

          " 1. Ambil data Vendor dan Bank

          READ TABLE lt_items_parsed INTO DATA(ls_first_p) INDEX 1.

          IF sy-subrc = 0.

            SELECT SINGLE lifnr FROM ekko INTO lv_vendor WHERE ebeln = ls_first_p-po_number.

            IF sy-subrc = 0.

              SELECT SINGLE name1 FROM lfa1 INTO lv_name1 WHERE lifnr = lv_vendor.

              

              SELECT SINGLE b~banka, k~bankn, k~koinh

                INTO (@lv_banka, @lv_bankn, @lv_koinh)

                FROM lfbk AS k INNER JOIN bnka AS b ON k~banks = b~banks AND k~bankl = b~bankl

                WHERE k~lifnr = @lv_vendor.

            ENDIF.

          ENDIF.



          SHIFT lv_vendor LEFT DELETING LEADING '0'.

          IF lv_banka IS INITIAL.

            lv_banka = 'No bank data'.

            lv_bankn = '-'.

            lv_koinh = '-'.

          ENDIF.



          " 2. Susun Data Material Items UI

          DATA lv_counter TYPE i VALUE 1.

          LOOP AT lt_items_parsed ASSIGNING FIELD-SYMBOL(<ls_parsed>).

            CLEAR ls_ui_item.

            ls_ui_item-itm = |{ lv_counter ALIGN = RIGHT WIDTH = 6 PAD = '0' }|.

            ls_ui_item-po_number = <ls_parsed>-po_number.

            ls_ui_item-value_num = <ls_parsed>-item_amount.

            

            " Fetch Material Number

            SELECT SINGLE matnr FROM ekpo INTO ls_ui_item-material

              WHERE ebeln = <ls_parsed>-po_number AND ebelp = <ls_parsed>-po_item.

            SHIFT ls_ui_item-material LEFT DELETING LEADING '0'.

            

            " Format String currency

            WRITE ls_ui_item-value_num TO ls_ui_item-value_str CURRENCY lv_waers.

            CONDENSE ls_ui_item-value_str.

            

            lv_items_tot = lv_items_tot + ls_ui_item-value_num.

            APPEND ls_ui_item TO lt_ui_items.

            lv_counter = lv_counter + 1.

          ENDLOOP.



          " 3. Format Tanggal

          lv_invdt_str   = lv_inv_date+6(2) && '.' && lv_inv_date+4(2) && '.' && lv_inv_date(4).

          lv_pstngdt_str = lv_pstng_date+6(2) && '.' && lv_pstng_date+4(2) && '.' && lv_pstng_date(4).

          lv_duedt_str   = lv_bsl_date+6(2) && '.' && lv_bsl_date+4(2) && '.' && lv_bsl_date(4).



          " 4. Financial Summary

          lv_nett = lv_items_tot + lv_tax_amt.

          WRITE lv_items_tot TO lv_items_str CURRENCY lv_waers. CONDENSE lv_items_str.

          WRITE lv_tax_amt TO lv_tax_str CURRENCY lv_waers. CONDENSE lv_tax_str.

          WRITE lv_nett TO lv_nett_str CURRENCY lv_waers. CONDENSE lv_nett_str.



        ELSE.

          CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.

          lv_success = abap_false.

          lv_errmsg  = lv_msg.

        ENDIF.



      ENDIF.

    ENDIF.



  ENDIF.



  "========== SIDEBAR VARIABLES ==========

  DATA: lv_role TYPE string.

        lv_role = request->get_form_field( 'role' ).

        gv_active_page = ''.

    let currentScale = 1;

    function openModal(src) {

        document.getElementById('imgModal').classList.remove('hidden');

        const img = document.getElementById('modalImage');

        img.src = src;

        currentScale = 1;

        img.style.transform = `scale(${currentScale})`;

    }

    function closeModal() {

        document.getElementById('imgModal').classList.add('hidden');

        document.getElementById('modalImage').src = '';

    }

    function zoomIn() {

        currentScale += 0.25;

        document.getElementById('modalImage').style.transform = `scale(${currentScale})`;

    }

    function zoomOut() {

        if(currentScale > 0.3) currentScale -= 0.25;

        document.getElementById('modalImage').style.transform = `scale(${currentScale})`;

    }



    document.addEventListener("DOMContentLoaded", function() {

        var rawData = document.getElementById('rawAttachments').textContent.trim();

        if (rawData) {

            try {

                var docs = JSON.parse(rawData);

                var container = document.getElementById('attachments-container');



                docs.forEach(function(doc) {

                    var wrapper = document.createElement('div');

                    wrapper.className = 'border border-slate-200 rounded-lg overflow-hidden flex flex-col bg-white hover:shadow-md transition-shadow';



                    var header = document.createElement('div');

                    header.className = 'bg-slate-50 px-3 py-2 border-b border-slate-200 font-semibold text-xs text-slate-700 flex justify-between items-center';



                    var docTitle = doc.type;

                    if(doc.type === 'INV') docTitle = '1. Dokumen Invoice';

                    if(doc.type === 'PO')  docTitle = '2. Purchase Order';

                    if(doc.type === 'GR')  docTitle = '3. Goods Receipt';

                    if(doc.type === 'TAX') docTitle = '4. Faktur Pajak';



                    header.innerHTML = `<span>${docTitle}</span><span class="font-normal text-slate-500 bg-white px-1.5 py-0.5 border border-slate-200 rounded truncate max-w-[120px]" title="${doc.name}">${doc.name}</span>`;

                    wrapper.appendChild(header);



                    var body = document.createElement('div');

                    body.className = 'bg-slate-100 flex-1 p-2 flex items-center justify-center h-[250px] overflow-hidden relative group cursor-pointer';



                    var fileSrc = "data:" + doc.mime + ";base64," + doc.content;



                    if (doc.mime.includes('image')) {

                        body.onclick = function() { openModal(fileSrc); };

                        body.innerHTML = `<img src="${fileSrc}" class="w-full h-full object-cover group-hover:scale-105 transition-transform duration-300" alt="${docTitle}"/>

                                          <div class="absolute inset-0 bg-black/0 group-hover:bg-black/10 flex items-center justify-center transition-colors">

                                            <span class="opacity-0 group-hover:opacity-100 bg-black/60 text-white text-xs py-1 px-3 rounded-full backdrop-blur transition-opacity">Click to Zoom</span>

                                          </div>`;

                    } else if (doc.mime.includes('pdf')) {

                        body.innerHTML = `<object data="${fileSrc}" type="application/pdf" class="w-full h-full">

                                            <p class="text-xs text-slate-500 p-4 text-center">Preview tidak didukung. <br/><a href="${fileSrc}" download="${doc.name}" class="text-emerald-600 font-semibold underline mt-2 inline-block">Download PDF</a></p>

                                          </object>`;

                    } else {

                    }

                    container.appendChild(wrapper);

                });

                console.error("Gagal mem-parsing lampiran JSON.", e);

            }

        }

    });
