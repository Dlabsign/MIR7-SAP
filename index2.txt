        DELETE ADJACENT DUPLICATES FROM lt_lfbnr_keys.
        SELECT lblni AS lfbnr, xblnr
          FROM essr
          FOR ALL ENTRIES IN @lt_lfbnr_keys
          WHERE lblni = @lt_lfbnr_keys-table_line
          INTO CORRESPONDING FIELDS OF TABLE @lt_essr.
        SORT lt_essr BY lfbnr.

        LOOP AT lt_outstanding ASSIGNING FIELD-SYMBOL(<ls_ses_fill>).
          IF <ls_ses_fill>-lfbnr IS NOT INITIAL.
            READ TABLE lt_essr ASSIGNING FIELD-SYMBOL(<ls_essr_row>)
              WITH KEY lfbnr = <ls_ses_fill>-lfbnr BINARY SEARCH.
            IF sy-subrc = 0.
              <ls_ses_fill>-ses_xblnr = <ls_essr_row>-xblnr.
            ENDIF.
          ENDIF.
        ENDLOOP.
      ENDIF.
    ENDIF.

    SORT lt_outstanding BY lifnr name1 bedat ASCENDING ebeln ebelp lfbnr.
    lv_list_count = lines( lt_outstanding ).

    " ---- Build vendor stats ----
    DATA: lv_vs_lifnr_prev TYPE ekko-lifnr,
          lv_vs_idx        TYPE i,
          ls_vstat         TYPE ty_vendor_stats.
    CLEAR: lv_vs_lifnr_prev, lv_vs_idx.

    LOOP AT lt_outstanding ASSIGNING FIELD-SYMBOL(<ls_vs>).
      IF <ls_vs>-lifnr <> lv_vs_lifnr_prev.
        IF lv_vs_lifnr_prev IS NOT INITIAL.
          APPEND ls_vstat TO lt_vnd_stats.
        ENDIF.
        lv_vs_idx = lv_vs_idx + 1.
        CLEAR ls_vstat.
        ls_vstat-lifnr    = <ls_vs>-lifnr.
        ls_vstat-idx      = lv_vs_idx.
        ls_vstat-item_cnt = 0.
        ls_vstat-po_cnt   = 0.
        ls_vstat-val_sum  = 0.
        lv_vs_lifnr_prev  = <ls_vs>-lifnr.
      ENDIF.
      ls_vstat-item_cnt = ls_vstat-item_cnt + 1.
      ls_vstat-val_sum  = ls_vstat-val_sum + <ls_vs>-val_open.
    ENDLOOP.
    IF lv_vs_lifnr_prev IS NOT INITIAL.
      APPEND ls_vstat TO lt_vnd_stats.
    ENDIF.

    SORT lt_outstanding BY lifnr ebeln.
    DATA: lv_cnt_lifnr_prev TYPE ekko-lifnr,
          lv_cnt_ebeln_prev TYPE ekko-ebeln.
    CLEAR: lv_cnt_lifnr_prev, lv_cnt_ebeln_prev.

    LOOP AT lt_outstanding ASSIGNING FIELD-SYMBOL(<ls_vpo>).
      IF <ls_vpo>-lifnr <> lv_cnt_lifnr_prev OR <ls_vpo>-ebeln <> lv_cnt_ebeln_prev.
        IF <ls_vpo>-ebeln <> lv_cnt_ebeln_prev.
          READ TABLE lt_vnd_stats ASSIGNING FIELD-SYMBOL(<ls_vsc2>)
            WITH KEY lifnr = <ls_vpo>-lifnr.
          IF sy-subrc = 0.
            <ls_vsc2>-po_cnt = <ls_vsc2>-po_cnt + 1.
          ENDIF.
          lv_cnt_ebeln_prev = <ls_vpo>-ebeln.
        ENDIF.
        lv_cnt_lifnr_prev = <ls_vpo>-lifnr.
      ENDIF.
    ENDLOOP.

    SORT lt_outstanding BY lifnr name1 bedat ASCENDING ebeln ebelp lfbnr.

    " ---- Build PO stats ----
    DATA: lv_ps_ebeln_prev TYPE ekko-ebeln,
          lv_ps_lifnr_prev TYPE ekko-lifnr,
          lv_ps_po_idx     TYPE i,
          ls_pstat         TYPE ty_po_stats.
    CLEAR: lv_ps_ebeln_prev, lv_ps_lifnr_prev, lv_ps_po_idx.

    LOOP AT lt_outstanding ASSIGNING FIELD-SYMBOL(<ls_ps>).
      IF <ls_ps>-lifnr <> lv_ps_lifnr_prev.
        lv_ps_po_idx     = 0.
        lv_ps_lifnr_prev = <ls_ps>-lifnr.
      ENDIF.
      IF <ls_ps>-ebeln <> lv_ps_ebeln_prev.
        IF lv_ps_ebeln_prev IS NOT INITIAL.
          APPEND ls_pstat TO lt_po_stats.
        ENDIF.
        lv_ps_po_idx = lv_ps_po_idx + 1.
        CLEAR ls_pstat.
        ls_pstat-ebeln   = <ls_ps>-ebeln.
        ls_pstat-val_sum = 0.
        ls_pstat-idx     = lv_ps_po_idx.
        lv_ps_ebeln_prev = <ls_ps>-ebeln.
      ENDIF.
      ls_pstat-val_sum = ls_pstat-val_sum + <ls_ps>-val_open.
    ENDLOOP.
    IF lv_ps_ebeln_prev IS NOT INITIAL.
      APPEND ls_pstat TO lt_po_stats.
    ENDIF.

    lv_total_vendors = lines( lt_vnd_stats ).

  ENDIF.

  "========== FLOW LOGIC - SEARCH PO MANUAL ==========
  IF lv_action = 'search' AND lv_po_number IS NOT INITIAL.

    DATA: ls_out       TYPE ty_po_item,
          lv_gr_count  TYPE i,
          lv_po_exists TYPE abap_bool.

    SELECT SINGLE lifnr INTO @lv_vendor
      FROM ekko WHERE ebeln = @lv_po_number.

    IF sy-subrc = 0.
      lv_po_exists = abap_true.

      SELECT ebelp AS po_item, txz01 AS short_text, menge AS quantity,
             netpr AS net_price, werks AS plant
        FROM ekpo
        WHERE ebeln = @lv_po_number
        INTO CORRESPONDING FIELDS OF TABLE @lt_items.

      SELECT COUNT(*) INTO @lv_gr_count
        FROM ekbe
        WHERE ebeln = @lv_po_number
          AND bewtp = 'E'
          AND bwart = '101'.

      lv_gr_done = COND #( WHEN lv_gr_count > 0
                            THEN abap_true
                            ELSE abap_false ).

      IF lv_gr_done = abap_true.
        lv_gr_text = 'Sudah Diterima (GR OK)'.
      ELSE.
        lv_gr_text = 'Belum Ada GR'.
      ENDIF.

      lv_success = abap_true.
    ELSE.
      lv_errmsg = |Nomor PO { lv_po_number } tidak ditemukan di sistem.|.
      lv_success = abap_false.
    ENDIF.

  ENDIF.

    INTO TABLE lt_skat
    FROM skat
    WHERE spras = sy-langu
      AND ktopl = '1000'.

/* ======================================================================
   CSRF TOKEN
   ====================================================================== */
function getCookie(name) {
  var cookies = document.cookie.split(';');
  for (var i = 0; i < cookies.length; i++) {
    var c = cookies[i].trim();
    if (c.indexOf(name + '=') === 0) return c.substring(name.length + 1);
  }
  return '';
}
function getCsrfToken() {
  var token = getCookie('~csrf_token');
  if (!token) token = getCookie('sap-xsrf-token');
  if (!token) token = getCookie('X-CSRF-Token');
  return token;
}

   UPLOAD DOKUMEN (PRE-PARK MODAL)

function openUploadModal() {
  var chk = document.querySelectorAll('.item-cb:checked');
  if (chk.length === 0) return;

  // Reset form upload
  document.getElementById('file_inv').value = '';
  document.getElementById('file_gr').value  = '';
  document.getElementById('file_po').value  = '';

  document.getElementById('uploadErrorMsg').style.display = 'none';

  document.getElementById('uploadModal').classList.add('open');
}

  document.getElementById('uploadModal').classList.remove('open');

async function validateAndUploadDocs() {
  var fileInv = document.getElementById('file_inv').files[0];
  var fileGr  = document.getElementById('file_gr').files[0];
  var filePo  = document.getElementById('file_po').files[0];
  var fileTax = document.getElementById('file_tax').files[0];
  var errMsg  = document.getElementById('uploadErrorMsg');

  // VALIDASI DOKUMEN WAJIB
  if (!fileInv || !fileGr || !filePo) {
    errMsg.style.display = 'flex';
    return;
  }
  errMsg.style.display = 'none';

  var btn = document.getElementById('btnProcessUpload');
  btn.disabled = true;
  btn.innerHTML = '⏳ Sedang Mengunggah...';

  // FUNGSI 1: Konversi file mentah ke Base64 (untuk PDF / file non-gambar)
  const toBase64 = file => new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = () => resolve(reader.result.split(',')[1]);
    reader.onerror = error => reject(error);
  });

  // FUNGSI 2: Kompresi khusus gambar menggunakan Canvas
  // maxWidth mengatur lebar maksimal gambar, quality mengatur tingkat kompresi (0.0 - 1.0)
  const compressImage = (file, maxWidth = 1200, quality = 0.7) => new Promise((resolve, reject) => {
    // Jika file bukan gambar (misal PDF), langsung konversi ke Base64 tanpa kompresi
    if (!file.type.startsWith('image/')) {
       return toBase64(file).then(resolve).catch(reject);
    }

    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = event => {
      const img = new Image();
      img.src = event.target.result;
      img.onload = () => {
        const canvas = document.createElement('canvas');
        let width = img.width;
        let height = img.height;

        // Kalkulasi rasio aspek jika lebar melebihi batas maksimal
        if (width > maxWidth) {
          height = Math.round((height * maxWidth) / width);
          width = maxWidth;
        }

        canvas.width = width;
        canvas.height = height;
        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, width, height);

        // Render kembali ke Data URL dengan kompresi
        const mimeType = file.type === 'image/png' ? 'image/png' : 'image/jpeg';
        const compressedDataUrl = canvas.toDataURL(mimeType, quality);

        // Return hanya string Base64-nya saja (tanpa header 'data:image/jpeg;base64,')
        resolve(compressedDataUrl.split(',')[1]);
      };
      img.onerror = error => reject(error);
    };
    reader.onerror = error => reject(error);
  });

  try {
    var docsToUpload = [];

    // Ganti pemanggilan toBase64() menjadi compressImage()
    docsToUpload.push({ type: 'INV', name: fileInv.name, mime: fileInv.type, content: await compressImage(fileInv) });
    docsToUpload.push({ type: 'GR',  name: fileGr.name,  mime: fileGr.type,  content: await compressImage(fileGr) });
    docsToUpload.push({ type: 'PO',  name: filePo.name,  mime: filePo.type,  content: await compressImage(filePo) });
    if (fileTax) {
      docsToUpload.push({ type: 'TAX', name: fileTax.name, mime: fileTax.type, content: await compressImage(fileTax) });
    }

    // Ambil nomor PO pertama sebagai referensi
    var chk = document.querySelectorAll('.item-cb:checked');
    var refPo = chk[0].getAttribute('data-ebeln');

    var formData = new FormData();
    formData.append('action', 'upload_docs');
    formData.append('ref_doc', refPo);
    formData.append('docs_json', JSON.stringify(docsToUpload));
    formData.append('~csrf_token', getCsrfToken());

    // DEBUG 1: Gunakan window.location.href agar dinamis mengarah ke halaman yang sedang dibuka
    let response = await fetch(window.location.href, {
      method: 'POST',
      body: formData
    });

    // DEBUG 2: Ambil response sebagai teks mentah terlebih dahulu untuk mengecek apakah SAP mengembalikan HTML Error
    let rawText = await response.text();
    console.log("Raw Response dari ABAP:", rawText);

    if (!response.ok) {
        throw new Error(`HTTP Error ${response.status}: Silakan cek tab Network (F12) untuk detailnya.`);
    }

    // DEBUG 3: Coba parse JSON secara manual
    let result = JSON.parse(rawText);

    if (result.status === 'success') {
      closeUploadModal();
      openParkModal();
    } else {
      alert('Gagal mengunggah dokumen: ' + result.message);
    }
  } catch (error) {
    // DEBUG 4: Tampilkan error asli di Alert dan Console
    alert('Terjadi kesalahan sistem:\n\n' + error.message);
    console.error("Detail Error Upload:", error);
  } finally {
    btn.disabled = false;
    btn.innerHTML = '⬆️ Upload & Lanjutkan';
  }
}

function syncVendorCb(vIdx) {
  var all  = document.querySelectorAll('#vendor_' + vIdx + ' .item-cb');
  var chk  = document.querySelectorAll('#vendor_' + vIdx + ' .item-cb:checked');
  var vcb  = document.getElementById('vcb_' + vIdx);
  if (!vcb) return;
  vcb.indeterminate = false;
  if (chk.length === 0)               { vcb.checked = false; }
  else if (chk.length === all.length) { vcb.checked = true; }
  else                                { vcb.checked = false; vcb.indeterminate = true; }
}

function onItemCheck() {
  var chk = document.querySelectorAll('.item-cb:checked');
  document.getElementById('parkCount').textContent = chk.length;
  var totalAmount = 0;
  var currency = chk.length > 0 ? (chk[0].getAttribute('data-waers') || 'IDR') : 'IDR';
  chk.forEach(function(cb) { totalAmount += parseFloat(cb.getAttribute('data-amount')) || 0; });
  document.getElementById('parkTotal').textContent = currency + ' ' + totalAmount.toLocaleString('id-ID', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  document.getElementById('parkBar').classList.toggle('visible', chk.length > 0);
  document.querySelectorAll('.accordion-vendor').forEach(function(v) {
    var vIdx = v.id.replace('vendor_', '');
    syncVendorCb(vIdx);
    v.querySelectorAll('.po-group-cb').forEach(function(pocb) {
      var pk      = pocb.id.replace('pocb_', '');
      var items   = v.querySelectorAll('[data-po="' + pk + '"]');
      var checked = v.querySelectorAll('[data-po="' + pk + '"]:checked');
      pocb.indeterminate = false;
      if (checked.length === 0)               { pocb.checked = false; }
      else if (checked.length === items.length) { pocb.checked = true; }
      else                                      { pocb.checked = false; pocb.indeterminate = true; }
    });
  });
}

function clearAllSelection() {
  document.querySelectorAll('.item-cb:checked').forEach(function(cb)    { cb.checked = false; });

  onItemCheck();
}

function buildTaxOptions(selected) {
  var opts = [
    { v: '',   l: '— (none)' },
    { v: 'V0', l: 'V0 0%'   },
    { v: 'V1', l: 'V1 11%'  },
    { v: 'I0', l: 'I0 0%'   },
    { v: 'I1', l: 'I1 1%'   },
    { v: 'I2', l: 'I2 10%'  },
    { v: 'I3', l: 'I3 11%'  },
    { v: 'I4', l: 'I4 12%'  },
    { v: 'O0', l: 'O0 0%'   },
    { v: 'O1', l: 'O1 0%X'  },
    { v: 'O2', l: 'O2 10%'  },
    { v: 'O3', l: 'O3 11%'  },
    { v: 'O4', l: 'O4 12%'  }
  ];
  return opts.map(function(o) {
    return '<option value="' + o.v + '"' + (o.v === selected ? ' selected' : '') + '>' + o.l + '</option>';
  }).join('');
}

/* Pengecekan Qty & Amount Saat Diedit (GR Blocking Logic) */
function onItemFieldChange(el) {
  var key   = el.getAttribute('data-key');
  var field = el.getAttribute('data-field');
  if (!itemsData[key]) itemsData[key] = {};

  if (field === 'qty') {
    var rawQty = el.value.replace(/\./g, '').replace(',', '.');
    var parsedQty = parseFloat(rawQty) || 0;
    itemsData[key].qty = parsedQty;

    // Partial GR Blocking Check
    var maxQty = parseFloat(el.getAttribute('data-maxqty')) || 0;
    if (parsedQty > maxQty) {
      el.style.borderColor = '#ef4444';
      el.style.backgroundColor = '#fef2f2';
      el.title = 'Kuantitas Invoice melebihi Open Qty GR (Maks: ' + maxQty + ')';
    } else {
      el.style.borderColor = '';
      el.style.backgroundColor = '';
      el.title = '';
    }
  } else if (field === 'tax_code') {
    itemsData[key].tax_code = el.value;
  }
  updateItemsTotal();
  updateBalancePanel();
}

function onItemAmountInput(el) {
  var key = el.getAttribute('data-key');
  var raw = el.value.replace(/[^0-9.,]/g, '').replace(/\./g, '').replace(',', '.');
  var num = parseFloat(raw) || 0;
  if (!itemsData[key]) itemsData[key] = {};
  itemsData[key].amount = num;
  updateItemsTotal();
  updateBalancePanel();
}

function updateItemsTotal() {
  var total = 0;
  var chk   = document.querySelectorAll('.item-cb:checked');
  chk.forEach(function(cb) {
    var lfbnr = cb.getAttribute('data-lfbnr') || '';
    var key = cb.getAttribute('data-ebeln') + '_' + cb.getAttribute('data-ebelp') + (lfbnr ? '_' + lfbnr : '');
    if (itemsData[key] && itemsData[key].amount !== undefined) {
      total += itemsData[key].amount;
    } else {
      total += parseFloat(cb.getAttribute('data-amount')) || 0;
    }
  });
  var el = document.getElementById('itemsTotalVal');
  if (el) el.textContent = getCurrency() + ' ' + total.toLocaleString('id-ID', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
}

var ZTERM_DAYS = {
  '0001': 0,   '2000': 0,   'Z001': 1,
  '0002': 30,  'Z007': 7,
  '0003': 30,  'Z014': 14,
  '0008': 45,  'Z030': 30,
  'Z045': 45,  'Z060': 60,
  'Z090': 90,  'Z120': 120
};

function calcEndOfMonth(d) {
  return new Date(d.getFullYear(), d.getMonth() + 1, 0);
}

function onPaymentTermChange() {
  var sel    = document.getElementById('modal_pmnt_trms');
  var bslEl  = document.getElementById('modal_bsl_date');
  var hint   = document.getElementById('ztermDueDateHint');
  if (!sel || !hint) return;

  var zterm  = sel.value;
  var bslStr = bslEl ? bslEl.value : '';

  if (!zterm) { hint.textContent = ''; return; }
  if (!bslStr) { hint.textContent = 'Isi Baseline Date untuk melihat estimasi jatuh tempo.'; hint.style.color = '#94a3b8'; return; }

  var bslDate = new Date(bslStr + 'T00:00:00');
  var dueDate = null;
  var dueNote = '';

  if (ZTERM_DAYS[zterm] !== undefined) {
    dueDate = new Date(bslDate);
    dueDate.setDate(dueDate.getDate() + ZTERM_DAYS[zterm]);
  } else if (zterm === '0004' || zterm === '0006') {
    dueDate = calcEndOfMonth(bslDate);
  } else if (zterm === '0005') {
    dueDate = new Date(bslDate.getFullYear(), bslDate.getMonth() + 1, 10);
  } else if (zterm === '0007') {
    var d15 = new Date(bslDate.getFullYear(), bslDate.getMonth() + 1, 15);
    dueDate  = d15;
    dueNote  = ' (tgl 15 bln berikutnya — cek Day Limit)';
  } else if (zterm === '0009') {
    hint.textContent = 'ℹ️ Payable in 3 installments — jatuh tempo dihitung SAP per cicilan.';
    hint.style.color = '#64748b';
    return;
  }

  if (dueDate) {
    var dd  = String(dueDate.getDate()).padStart(2,'0');
    var mm  = String(dueDate.getMonth()+1).padStart(2,'0');
    var yy  = dueDate.getFullYear();
    hint.textContent = '📅 Est. Jatuh Tempo: ' + dd + '/' + mm + '/' + yy + dueNote;
    hint.style.color = '#059669';
  } else {
    hint.textContent = '';
  }
}

/* ======================================================================
   BALANCE PANEL
   ====================================================================== */
function updateBalancePanel() {
  var grossRaw = (document.getElementById('modal_gross_amount').value || '').replace(/[^0-9.,]/g, '').replace(/\./g, '').replace(',', '.');
  var gross    = parseFloat(grossRaw) || 0;
  var calcTaxOn= (document.getElementById('modal_calc_tax')   || {}).checked;

  var chk      = document.querySelectorAll('.item-cb:checked');
  var sumItems = 0;
  var taxEst   = 0;

  chk.forEach(function(cb) {
    var lfbnr  = cb.getAttribute('data-lfbnr') || '';
    var key    = cb.getAttribute('data-ebeln') + '_' + cb.getAttribute('data-ebelp') + (lfbnr ? '_' + lfbnr : '');
    var amt    = (itemsData[key] && itemsData[key].amount !== undefined)
                 ? itemsData[key].amount
                 : (parseFloat(cb.getAttribute('data-amount')) || 0);

    var itemTax = (itemsData[key] && itemsData[key].tax_code !== undefined)
                  ? itemsData[key].tax_code
                  : ((document.getElementById('modal_tax_code') || {}).value || '');

    var rate = TAX_RATES[itemTax] !== undefined ? TAX_RATES[itemTax] : 0;

    sumItems += amt;
    if (calcTaxOn && rate > 0) {
      taxEst += amt * rate;
    }
  });

  var taxAmtEl = document.getElementById('modal_tax_amount');
  var taxUsed  = 0;
  if (calcTaxOn) {
    if (taxAmtEl) taxAmtEl.value = taxEst > 0 ? taxEst.toString() : '';
    taxUsed = taxEst;
  } else {
    var taxManualRaw = (taxAmtEl ? taxAmtEl.value : '').replace(/[^0-9.,]/g, '').replace(/\./g, '').replace(',', '.');
    taxUsed = parseFloat(taxManualRaw) || 0;
  }

  var expectedGross = sumItems + taxUsed;
  var balance       = gross - expectedGross;

  // Set global balance untuk validasi
  window.currentBalance = balance;

  document.getElementById('bp_gross').textContent = gross    > 0 ? fmtIDR(gross)   : '—';
  document.getElementById('bp_items').textContent = sumItems > 0 ? fmtIDR(sumItems) : (gross > 0 ? 'IDR 0' : '—');
  document.getElementById('bp_tax').textContent   = taxUsed  > 0 ? fmtIDR(taxUsed)  : (gross > 0 ? 'IDR 0' : '—');

  var row    = document.getElementById('balanceStatusRow');
  var dot    = document.getElementById('balanceDot');
  var statusTx = document.getElementById('balanceStatusText');
  var balVal   = document.getElementById('balanceVal');

  if (gross <= 0) {
    row.className        = 'balance-status-pill idle';
    dot.className        = 'balance-dot idle';
    statusTx.textContent = sumItems > 0 ? 'Expected: ' + fmtIDR(expectedGross) : '—';
    balVal.textContent   = '';
  } else if (Math.abs(balance) <= 1) {
    row.className        = 'balance-status-pill ok';
    dot.className        = 'balance-dot ok';
    statusTx.textContent = 'BALANCE';
    balVal.textContent   = '';
  } else {
    row.className        = 'balance-status-pill err';
    dot.className        = 'balance-dot err';
    statusTx.textContent = balance > 0 ? '▲ ' + fmtIDR(Math.abs(balance)) : '▼ ' + fmtIDR(Math.abs(balance));
    balVal.textContent   = '';
  }

  validateParkForm();
}

/* ======================================================================
   FORM VALIDATION (ENFORCING ZERO BALANCE)
   ====================================================================== */
function validateParkForm() {
  var btn = document.getElementById('btnParkInvoice');
  if (!btn) return;
  var invDate   = (document.getElementById('modal_inv_date')      || {}).value || '';
  var pstngDate = (document.getElementById('modal_pstng_date')    || {}).value || '';
  var refDoc    = ((document.getElementById('modal_ref_doc')      || {}).value || '').trim();
  var grossRaw  = ((document.getElementById('modal_gross_amount') || {}).value || '').replace(/[^0-9.,]/g, '').replace(/\./g, '').replace(',', '.');
  var grossNum  = parseFloat(grossRaw) || 0;

  var checks = {
    inv_date:     !!invDate,
    pstng_date:   !!pstngDate,
    ref_doc:      refDoc.length > 0,
    gross_amount: grossNum > 0,
    balance_zero: Math.abs(window.currentBalance || 0) <= 1 // Toleransi pembulatan 1 Rupiah
  };

  function markField(id, ok) {
    var el = document.getElementById(id);
    if (!el) return;
    el.classList.toggle('field-error', !ok && el.value !== '');
    el.classList.toggle('field-ok',    ok);
  }
  markField('modal_inv_date',     checks.inv_date);
  markField('modal_pstng_date',   checks.pstng_date);
  markField('modal_ref_doc',      checks.ref_doc);
  markField('modal_gross_amount', checks.gross_amount);

  var allOk   = checks.inv_date && checks.pstng_date && checks.ref_doc && checks.gross_amount && checks.balance_zero;
  btn.disabled = !allOk;

  if (!checks.inv_date)        btn.title = 'Invoice Date wajib diisi';
  else if (!checks.pstng_date) btn.title = 'Posting Date wajib diisi';
  else if (!checks.ref_doc)    btn.title = 'Reference Document No. wajib diisi';
  else if (!checks.gross_amount) btn.title = 'Amount wajib diisi';
  else if (!checks.balance_zero) btn.title = 'Balance belum NOL. Sesuaikan Gross Amount atau Value per Item/GL.';
  else                         btn.title = '';
}

var PER_PAGE    = 5;
var currentPage = 1;

function getFilterValues() {
  var q = (document.getElementById('filterUnified') ? document.getElementById('filterUnified').value : '').toLowerCase().trim();
  return { q: q };
}
function isFiltering() { return getFilterValues().q.length > 0; }
function itemMatches(row, f) {
  if (!f.q) return true;
  var ebeln     = (row.getAttribute('data-ebeln')      || '').toLowerCase();
  var ebelp     = (row.getAttribute('data-ebelp')      || '').toLowerCase();
  var bedat     = (row.getAttribute('data-bedat')      || '').toLowerCase();
  var txz01     = (row.getAttribute('data-txz01')      || '').toLowerCase();
  var mblnr     = (row.getAttribute('data-mblnr')      || '').toLowerCase();
  var lfbnr     = (row.getAttribute('data-lfbnr')      || '').toLowerCase();
  var sesXblnr  = (row.getAttribute('data-ses-xblnr')  || '').toLowerCase();
  var mblnrStripped    = mblnr.replace(/^0+/, '');
  var lfbnrStripped    = lfbnr.replace(/^0+/, '');
  var sesXblnrStripped = sesXblnr.replace(/^0+/, '');
  return (ebeln.indexOf(f.q) >= 0    || ebelp.indexOf(f.q) >= 0 ||
          bedat.indexOf(f.q) >= 0    || txz01.indexOf(f.q) >= 0 ||
          mblnr.indexOf(f.q) >= 0    || mblnrStripped.indexOf(f.q) >= 0 ||
          lfbnr.indexOf(f.q) >= 0    || lfbnrStripped.indexOf(f.q) >= 0 ||
          sesXblnr.indexOf(f.q) >= 0 || sesXblnrStripped.indexOf(f.q) >= 0);
}
function vendorMatches(vendor, f) {
  if (!f.q) return true;
  var name  = (vendor.getAttribute('data-vendor-name')  || '').toLowerCase();
  var lifnr = (vendor.getAttribute('data-vendor-lifnr') || '').toLowerCase();
  return (name.indexOf(f.q) >= 0 || lifnr.indexOf(f.q) >= 0);
}
function filterData() {
  var f          = getFilterValues();
  var filtering  = isFiltering();
  var clearBtn   = document.getElementById('filterClearBtn');
  if (clearBtn) {
    if (filtering) clearBtn.classList.add('visible');
    else           clearBtn.classList.remove('visible');
  }
  var vendors       = document.querySelectorAll('.accordion-vendor');
  var visibleItems  = 0;
  vendors.forEach(function(vendor) {
    var vMatch      = vendorMatches(vendor, f);
    var poGroups    = vendor.querySelectorAll('.po-group');
    var vendorHasMatch = false;
    poGroups.forEach(function(poGroup) {
      var groupRows    = poGroup.querySelectorAll('.po-item-row');
      var groupHasMatch = false;
      groupRows.forEach(function(row) {
        var show = vMatch || itemMatches(row, f);
        if (show) { row.style.display = ''; groupHasMatch = true; visibleItems++; }
        else      { row.style.display = 'none'; }
      });
      poGroup.style.display = groupHasMatch ? '' : 'none';
      if (groupHasMatch) vendorHasMatch = true;
    });
    if (vendorHasMatch) vendor.classList.remove('filter-hidden');
    else                vendor.classList.add('filter-hidden');
    if (filtering && vendorHasMatch) vendor.classList.add('open');
    if (!filtering)                  vendor.classList.remove('open');
  });
  var si = document.getElementById('summaryVisible');
  if (si) si.textContent = visibleItems;
  currentPage = 1;
  applyPagination();
}
function clearFilters() {
  var el = document.getElementById('filterUnified');
  if (el) el.value = '';
  var clearBtn = document.getElementById('filterClearBtn');
  if (clearBtn) clearBtn.classList.remove('visible');
  filterData();
}

function applyPagination() {
  var vendors        = document.querySelectorAll('.accordion-vendor');
  var visibleVendors = [];
  vendors.forEach(function(v) {
    if (!v.classList.contains('filter-hidden')) visibleVendors.push(v);
  });
  var totalPages = Math.ceil(visibleVendors.length / PER_PAGE) || 1;
  if (currentPage > totalPages) currentPage = totalPages;
  var startIdx = (currentPage - 1) * PER_PAGE;
  var endIdx   = startIdx + PER_PAGE;
  visibleVendors.forEach(function(v, i) {
    v.style.display = (i >= startIdx && i < endIdx) ? '' : 'none';
  });
  vendors.forEach(function(v) {
    if (v.classList.contains('filter-hidden')) v.style.display = 'none';
  });
  renderPagination(totalPages, visibleVendors.length);
}
function renderPagination(totalPages) {
  var bar = document.getElementById('paginationBar');
  if (!bar) return;
  if (totalPages <= 1) { bar.innerHTML = ''; return; }
  var html = '';
  html += currentPage > 1
    ? '<a href="javascript:void(0)" class="page-btn" onclick="goPage(' + (currentPage-1) + ')">&#9664; Sebelumnya</a>'
    : '<span class="page-btn disabled">&#9664; Sebelumnya</span>';
  var startP = Math.max(1, currentPage - 2);
  var endP   = Math.min(totalPages, startP + 4);
  if (endP - startP < 4) startP = Math.max(1, endP - 4);
  for (var i = startP; i <= endP; i++) {
    html += i === currentPage
      ? '<span class="page-btn active">' + i + '</span>'
      : '<a href="javascript:void(0)" class="page-btn" onclick="goPage(' + i + ')">' + i + '</a>';
  }
  html += currentPage < totalPages
    ? '<a href="javascript:void(0)" class="page-btn" onclick="goPage(' + (currentPage+1) + ')">Berikutnya &#9654;</a>'
    : '<span class="page-btn disabled">Berikutnya &#9654;</span>';
  bar.innerHTML = html;
}
function goPage(p) {
  currentPage = p;
  applyPagination();
  var pb = document.getElementById('paginationBar');
  if (pb) window.scrollTo({ top: pb.offsetTop - 120, behavior: 'smooth' });
}

<%
  DATA: lv_saknr TYPE string,
        lv_txt   TYPE string,

  lv_sep = ''.
  LOOP AT lt_skat ASSIGNING FIELD-SYMBOL(<ls_skat>).

    " Bersihkan karakter kutip/backslash agar tidak merusak syntax JavaScript
    lv_txt = <ls_skat>-txt50.
    REPLACE ALL OCCURRENCES OF `\` IN lv_txt WITH `\\`.
    REPLACE ALL OCCURRENCES OF `'` IN lv_txt WITH `\'`.
%>
  <%= lv_sep %>['<%= lv_saknr %>', '<%= lv_txt %>']
<%
    lv_sep = ','.
  ENDLOOP.
%>
];

function searchGL(q) {
  if (!q || q.length < 2) return [];
  var ql = q.toLowerCase();
  var results = [];
  for (var i = 0; i < GL_MASTER.length; i++) {
    var code = GL_MASTER[i][0];
    var name = GL_MASTER[i][1];
    if (code.toLowerCase().indexOf(ql) === 0 || name.toLowerCase().indexOf(ql) >= 0) {
      results.push([code, name]);
      if (results.length >= 10) break;
    }
  }
  return results;
}

    // Hitung total DPP item untuk Auto-fill
    var totalDpp = 0;
    var chk = document.querySelectorAll('.item-cb:checked');
    chk.forEach(function(cb) {
      var lfbnr = cb.getAttribute('data-lfbnr') || '';
      var key = cb.getAttribute('data-ebeln') + '_' + cb.getAttribute('data-ebelp') + (lfbnr ? '_' + lfbnr : '');
      var amt = (itemsData[key] && itemsData[key].amount !== undefined)
                 ? itemsData[key].amount
                 : (parseFloat(cb.getAttribute('data-amount')) || 0);
      totalDpp += amt;
    });

    // Auto-fill ke W/Tax Base FC jika ada nilainya
    if (baseEl && totalDpp > 0) {
      baseEl.value = totalDpp.toLocaleString('id-ID', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
    }

  } else if (hint) {
    hint.style.display = 'none';
    if (baseEl) baseEl.value = '';
  }

  // Set default percentage & kalkulasi otomatis
  if (pctEl && WTAX_PCT_DEFAULTS[val] !== undefined) {
    pctEl.value = WTAX_PCT_DEFAULTS[val].toString().replace('.', ',');
    recalcWTaxRow(idx);
  } else {
    if (pctEl) pctEl.value = '';
    var amtEl = document.getElementById('wtax_amt_' + idx);
    if (amtEl) amtEl.value = '';
  }

  updateWTaxTotal();
}

function recalcWTaxRow(idx) {
  var baseEl  = document.getElementById('wtax_base_' + idx);
  var amtEl   = document.getElementById('wtax_amt_'  + idx);
  var pctEl   = document.getElementById('wtax_pct_'  + idx);
  if (!baseEl || !amtEl || !pctEl) return;
  var baseRaw = baseEl.value.replace(/\./g,'').replace(',','.');
  var pctRaw  = pctEl.value.replace(',','.');
  var base    = parseFloat(baseRaw) || 0;
  var pct     = parseFloat(pctRaw)  || 0;
  if (base > 0 && pct > 0) {
    var computed = base * pct / 100;
    amtEl.value  = computed.toLocaleString('id-ID', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }
}

function removeWTaxRow(idx) {
  var row = document.getElementById('wtax_row_' + idx);
  if (row) row.remove();
  updateWTaxTotal();
}

function updateWTaxTotal() {
  var rows  = getWTaxRows();
  var total = 0;
  rows.forEach(function(r) { total += r.wtax_amount; });
  var el = document.getElementById('wtaxTotalVal');
  if (el) el.textContent = 'IDR ' + total.toLocaleString('id-ID', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
}

function getWTaxRows() {
  var result = [];
  var cont   = document.getElementById('wtaxRows');
  if (!cont) return result;
  cont.querySelectorAll('[id^="wtax_row_"]').forEach(function(row) {
    var idx     = row.id.replace('wtax_row_', '');
    var selEl   = document.getElementById('wtax_type_' + idx);
    var baseEl  = document.getElementById('wtax_base_' + idx);
    var amtEl   = document.getElementById('wtax_amt_'  + idx);
    var pctEl   = document.getElementById('wtax_pct_'  + idx);
    if (!selEl || !selEl.value) return;
    var parts   = selEl.value.split('|');
    var baseRaw = (baseEl ? baseEl.value : '').replace(/\./g,'').replace(',','.');
    var amtRaw  = (amtEl  ? amtEl.value  : '').replace(/\./g,'').replace(',','.');
    var pctRaw  = (pctEl  ? pctEl.value  : '').replace(',','.');
    result.push({
      wtax_type:   parts[0] || '',
      wtax_code:   parts[1] || '',
      wtax_base:   parseFloat(baseRaw) || 0,
      wtax_amount: parseFloat(amtRaw)  || 0,
      wtax_pct:    parseFloat(pctRaw)  || 0
    });
  });
  return result;
}
