* ======================================================================
* parked_doc.htm — Parked Documents (View Only, No Actions)
* ======================================================================
  DATA: lv_role        TYPE string,
        gv_active_page TYPE string.

  lv_role        = request->get_form_field( 'role' ).
  gv_active_page = 'PARKED'.

  " Main Table & Supporting (MIRO / RBKP)
  DATA: it_parked_inv TYPE TABLE OF rbkp, wa_inv TYPE rbkp,
        it_rseg       TYPE TABLE OF rseg, wa_rseg TYPE rseg,
        it_rbco       TYPE TABLE OF rbco, wa_rbco TYPE rbco,
        it_lfa1       TYPE TABLE OF lfa1, wa_lfa1 TYPE lfa1,
        it_lfbk       TYPE TABLE OF lfbk, wa_lfbk TYPE lfbk,
        it_bnka       TYPE TABLE OF bnka, wa_bnka TYPE bnka,
        it_t052       TYPE TABLE OF t052, wa_t052 TYPE t052,
        it_attach     TYPE TABLE OF zapp_attachment, wa_attach TYPE zapp_attachment.

* ======================================================================
* DATA FETCH & FILTER
* ======================================================================
  SELECT * FROM rbkp INTO TABLE it_parked_inv
    WHERE rbstat IN ('A', 'B') AND blart = 'RE'
    ORDER BY cpudt DESCENDING.

  IF it_parked_inv IS NOT INITIAL.
    DATA: lv_tabix       TYPE sytabix,
          lv_latest_flag TYPE zapp_log-flag.

    " Hanya tampil jika belum di-log ATAU flag terakhir = 3 (failed/resubmit)
    LOOP AT it_parked_inv INTO wa_inv.
      lv_tabix = sy-tabix.
      CLEAR lv_latest_flag.

      SELECT flag FROM zapp_log INTO @lv_latest_flag
        UP TO 1 ROWS
        WHERE belnr = @wa_inv-belnr AND gjahr = @wa_inv-gjahr
        ORDER BY record DESCENDING.
      ENDSELECT.

      IF sy-subrc = 0 AND ( lv_latest_flag = 0 OR lv_latest_flag = 1 ).
        DELETE it_parked_inv INDEX lv_tabix.
      ENDIF.
    ENDLOOP.

    IF it_parked_inv IS NOT INITIAL.
      SELECT * FROM rseg INTO TABLE it_rseg FOR ALL ENTRIES IN it_parked_inv WHERE belnr = it_parked_inv-belnr AND gjahr = it_parked_inv-gjahr.
      SELECT * FROM rbco INTO TABLE it_rbco FOR ALL ENTRIES IN it_parked_inv WHERE belnr = it_parked_inv-belnr AND gjahr = it_parked_inv-gjahr.
      SELECT * FROM lfa1 INTO TABLE it_lfa1 FOR ALL ENTRIES IN it_parked_inv WHERE lifnr = it_parked_inv-lifnr.
      SELECT * FROM lfbk INTO TABLE it_lfbk FOR ALL ENTRIES IN it_parked_inv WHERE lifnr = it_parked_inv-lifnr.
      IF it_lfbk IS NOT INITIAL.
        SELECT * FROM bnka INTO TABLE it_bnka FOR ALL ENTRIES IN it_lfbk WHERE banks = it_lfbk-banks AND bankl = it_lfbk-bankl.
      ENDIF.
      SELECT * FROM t052 INTO TABLE it_t052 FOR ALL ENTRIES IN it_parked_inv WHERE zterm = it_parked_inv-zterm.

      " --- FETCH ATTACHMENTS (Diambil dari No. PO atau No. Invoice) ---
      DATA: lt_ref_keys TYPE TABLE OF zapp_attachment-ref_doc,
            lv_ref_key  TYPE zapp_attachment-ref_doc.

      LOOP AT it_parked_inv INTO wa_inv.
        lv_ref_key = wa_inv-belnr.
        APPEND lv_ref_key TO lt_ref_keys.
      ENDLOOP.
      LOOP AT it_rseg INTO wa_rseg.
        lv_ref_key = wa_rseg-ebeln.
        APPEND lv_ref_key TO lt_ref_keys.
      ENDLOOP.
      SORT lt_ref_keys.
      DELETE ADJACENT DUPLICATES FROM lt_ref_keys.

      IF lt_ref_keys IS NOT INITIAL.
        SELECT * FROM zapp_attachment INTO TABLE it_attach
          FOR ALL ENTRIES IN lt_ref_keys
          WHERE ref_doc = lt_ref_keys-table_line.
      ENDIF.
    ENDIF.
  ENDIF.
%>

        function toggleDetail(belnr) {
            const detailRow = document.getElementById('detail_' + belnr);
            const icon = document.getElementById('icon_' + belnr);
            if (detailRow.classList.contains('hidden')) {
                detailRow.classList.remove('hidden');
                icon.style.transform = 'rotate(180deg)';
            } else {
                detailRow.classList.add('hidden');
                icon.style.transform = 'rotate(0deg)';
            }
        }

        let isAllExpanded = false;
        function toggleAllDetails() {
            isAllExpanded = !isAllExpanded;
            const detailRows = document.querySelectorAll('tr[id^="detail_"]');
            const icons = document.querySelectorAll('svg[id^="icon_"]');
            const btn = document.getElementById('btnToggleAll');
            detailRows.forEach(row => {
                const mainRowId = row.id.replace('detail_', '');
                const mainRow = document.querySelector(`tr[data-belnr="${mainRowId}"]`);
                if (mainRow && mainRow.style.display !== 'none') {
                    if (isAllExpanded) row.classList.remove('hidden');
                    else row.classList.add('hidden');
                }
            });
            icons.forEach(icon => {
                const mainRowId = icon.id.replace('icon_', '');
                const mainRow = document.querySelector(`tr[data-belnr="${mainRowId}"]`);
                if (mainRow && mainRow.style.display !== 'none') {
                    if (isAllExpanded) icon.style.transform = 'rotate(180deg)';
                    else icon.style.transform = 'rotate(0deg)';
                }
            });
            btn.innerHTML = isAllExpanded
                ? '<svg class="w-4 h-4 mr-2" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 15l7-7 7 7"></path></svg> Collapse All'

        }

        function searchTable() {
            const input = document.getElementById('searchInput').value.toUpperCase();
            const rows = document.querySelectorAll('tr.main-row');
            rows.forEach(row => {
                const vendorData = row.getAttribute('data-vendor').toUpperCase();
                const belnr = row.getAttribute('data-belnr');
                const detailRow = document.getElementById('detail_' + belnr);
                if (vendorData.indexOf(input) > -1) {
                    row.style.display = '';
                } else {
                    row.style.display = 'none';
                    if (detailRow && !detailRow.classList.contains('hidden')) {
                        detailRow.classList.add('hidden');
                        const icon = document.getElementById('icon_' + belnr);
                        if (icon) icon.style.transform = 'rotate(0deg)';
                    }
                }
            });
        }

        function doLogout() {
            if (!confirm('Yakin ingin logout?')) return;
            sessionStorage.setItem('logged_out', '1');
            fetch('/sap/public/bc/icf/logoff', { credentials: 'include' })
                .finally(function() {
                    var xhr = new XMLHttpRequest();
                    xhr.open('GET', '/sap/bc/bsp/sap/ZMIR7_BSP_AP/index.htm', true, 'logout', 'logout');
                    xhr.onreadystatechange = function() {
                        if (xhr.readyState === 4) {
                            window.location.replace('/sap/bc/bsp/sap/ZMIR7_BSP_AP/index.htm');
                        }
                    };
                    xhr.send();
                });
        }
        window.addEventListener('pageshow', function(e) {
            if (e.persisted && sessionStorage.getItem('logged_out') === '1') {
                sessionStorage.removeItem('logged_out');
                window.location.replace('/sap/public/bc/icf/logoff');
            }
        });
